# 4. Ruby Implementation (Production)

## trailing_engine.rb

```ruby
module Orders
  class TrailingEngine
    NIFTY_CONFIG = {
      breakeven_trigger: 0.10,
      activation_trigger: 0.20,
      trailing_distance: 0.22,
      early_trigger: 0.05,
      early_sl: -0.12
    }.freeze

    SENSEX_CONFIG = {
      breakeven_trigger: 0.12,
      activation_trigger: 0.18,
      trailing_distance: 0.30,
      early_trigger: 0.06,
      early_sl: -0.10
    }.freeze

    def initialize(position:, ltp:, highest_price:)
      @position = position
      @ltp = ltp.to_f
      @highest = highest_price.to_f
      @entry = position.entry_price.to_f
      @symbol = position.symbol
    end

    def call
      config = config_for_symbol

      profit_pct = (@ltp - @entry) / @entry

      return early_stop(config) if profit_pct >= config[:early_trigger] && profit_pct < config[:breakeven_trigger]

      return breakeven if profit_pct >= config[:breakeven_trigger] && profit_pct < config[:activation_trigger]

      return trailing_stop(config) if profit_pct >= config[:activation_trigger]

      nil
    end

    private

    def config_for_symbol
      if @symbol.include?("SENSEX")
        SENSEX_CONFIG
      else
        NIFTY_CONFIG
      end
    end

    def early_stop(config)
      @entry * (1 + config[:early_sl])
    end

    def breakeven
      @entry
    end

    def trailing_stop(config)
      @highest * (1 - config[:trailing_distance])
    end
  end
end
```

# Production Ruby Implementation

### `app/services/orders/trailing_engine.rb`

```ruby
module Orders
  class TrailingEngine
    NIFTY = {
      early: 0.05,
      breakeven: 0.10,
      activate: 0.20,
      distance: 0.22
    }.freeze

    SENSEX = {
      early: 0.06,
      breakeven: 0.12,
      activate: 0.18,
      distance: 0.30
    }.freeze

    def initialize(position:, ltp:)
      @position = position
      @entry = position.entry_price.to_f
      @ltp = ltp.to_f
      @highest = position.highest_price || @entry
    end

    def call
      update_highest

      cfg = config

      profit = (@ltp - @entry) / @entry

      return early_stop(cfg) if phase_early?(profit, cfg)
      return breakeven if phase_breakeven?(profit, cfg)
      return trailing(cfg) if phase_trailing?(profit, cfg)

      nil
    end

    private

    def config
      if @position.symbol.include?("SENSEX")
        SENSEX
      else
        NIFTY
      end
    end

    def update_highest
      if @ltp > @highest
        @highest = @ltp
        @position.update_columns(highest_price: @highest)
      end
    end

    def phase_early?(profit, cfg)
      profit >= cfg[:early] && profit < cfg[:breakeven]
    end

    def phase_breakeven?(profit, cfg)
      profit >= cfg[:breakeven] && profit < cfg[:activate]
    end

    def phase_trailing?(profit, cfg)
      profit >= cfg[:activate]
    end

    def early_stop(cfg)
      @entry * 0.88
    end

    def breakeven
      @entry
    end

    def trailing(cfg)
      @highest * (1 - cfg[:distance])
    end
  end
end
```
